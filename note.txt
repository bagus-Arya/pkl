SELECT user.nm_user,usaha.nm_usaha, lokasi.lokasi 
FROM usaha  
INNER JOIN lokasi  
ON(usaha.id_lokasi = lokasi.id_lokasi)
INNER JOIN user 
ON(user.id_user = usaha.id_user)
WHERE user.id_user = 1
